## 実装例
```python
import numpy as np
import json

class VectorStore:
    def __init__(self, vector_dir, metadata_dir):
        self.vector_dir = vector_dir
        self.metadata_dir = metadata_dir
    
    def save(self, vectors, metadata):
        np.save(f"{self.vector_dir}/vectors.npy", vectors)
        with open(f"{self.metadata_dir}/metadata.json", 'w') as f:
            json.dump(metadata, f)
    
    def load(self):
        vectors = np.load(f"{self.vector_dir}/vectors.npy")
        with open(f"{self.metadata_dir}/metadata.json", 'r') as f:
            metadata = json.load(f)
        return vectors, metadata
```
