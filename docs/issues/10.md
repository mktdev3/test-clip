## 実装例
```python
import argparse
from src.clip_model import CLIPModel
from src.text_encoder import TextEncoder
from src.vector_store import VectorStore
from src.search import SearchEngine

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument('query', type=str, help='検索クエリ')
    parser.add_argument('--top-k', type=int, default=10)
    args = parser.parse_args()
    
    clip = CLIPModel("rinna/japanese-clip-vit-b-16")
    text_encoder = TextEncoder(clip)
    vector_store = VectorStore("data/vectors", "data/metadata")
    search_engine = SearchEngine(vector_store)
    
    query_vector = text_encoder.encode_text(args.query)
    results = search_engine.search(query_vector, args.top_k)
    
    for i, result in enumerate(results):
        print(f"{i+1}. {result['image_path']} (score: {result['score']:.4f})")

if __name__ == "__main__":
    main()
```
